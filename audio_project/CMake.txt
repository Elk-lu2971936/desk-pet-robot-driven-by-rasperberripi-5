# Original basic configuration
cmake_minimum_required(VERSION 3.5)
project(MyLocalWhisperDemo)

set(CMAKE_CXX_STANDARD 14)

# Find PortAudio
find_package(PkgConfig REQUIRED)
pkg_check_modules(PORTAUDIO REQUIRED portaudio-2.0)
find_package(Threads REQUIRED)

# Add executable file
add_executable(myRecorder
    main.cpp
    microphoneInitialize.cpp
    whisper/src/whisper.cpp
    whisper/ggml/src/ggml.c
    # whisper/k_quants.cpp     
)
# Link the PortAudio library
target_include_directories(myRecorder PRIVATE
    ${PORTAUDIO_INCLUDE_DIRS}
    ${CMAKE_CURRENT_SOURCE_DIR}/whisper/src   

target_link_libraries(myRecorder PRIVATE
    ${PORTAUDIO_LIBRARIES}
    Threads::Threads
)
